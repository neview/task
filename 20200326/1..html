(function (){

    String.prototype.unique = function unique(){

        let obj = {};

        for(let i=0;i<this.length-1;i++){

            let time = this[i];

            if(typeof obj[time] !== undefined){

                obj[time] = this[this.length-1];

                this.length --;

                i--; 

            }

            obj[time] = time;

        }

        obj = null;

        return this

    };

})()

(function(){

    String.prototype.getParam = function getParam (key){

        let paramIndex = this.indexOf("?"),

            paramtext = '';

        let obj;

        paramIndex !== -1 ? paramtext = this.toString(paramIndex+1):null;

        if(paramtext){

            let time = paramtext.split('&');

            for(let i=0;i<time.length;i++){

                let tim = time.split('=');

                tim[0] ===key ? obj = tim[1]:null;

            }

        };

        return obj;

    };  

})();
